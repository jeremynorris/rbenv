# File lib/puppetlabs_spec_helper/puppetlabs_spec/fixtures.rb, line 25
def my_fixture(name)
  file = File.join(my_fixture_dir, name)
  unless File.readable? file then
    fail "fixture '#{name}' for #{my_fixture_dir} is not readable"
  end
  return file
end
my_fixture_dir() click to toggle source

Returns the path to your relative fixture dir. So if your spec test is <project>/spec/unit/facter/foo_spec.rb then your relative dir will be <project>/spec/fixture/unit/facter/foo

# File lib/puppetlabs_spec_helper/puppetlabs_spec/fixtures.rb, line 14
def my_fixture_dir
  callers = caller
  while line = callers.shift do
    next unless found = line.match(%{/spec/(.*)_spec\.rb:})
    return fixtures(found[1])
  end
  fail "sorry, I couldn't work out your path from the caller stack!"
end

Provides a block mechanism for iterating across the files in your fixture area.

# File lib/puppetlabs_spec_helper/puppetlabs_spec/fixtures.rb, line 41
  def my_fixtures(glob = '*', flags = 0)
    files = Dir.glob(File.join(my_fixture_dir, glob), flags)
    unless files.length > 0 then
      fail "fixture '#{glob}' for #{my_fixture_dir} had no files!"
    end
    block_given? and files.each do |file| yield file end
    files
  end
end
