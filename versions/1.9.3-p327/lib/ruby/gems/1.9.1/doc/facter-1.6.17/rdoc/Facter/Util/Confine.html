def true?
  unless fact = Facter[@fact]
    Facter.debug "No fact for %s" % @fact
    return false
  end
  value = convert(fact.value)

  return false if value.nil?

  @values.each do |v|
    v = convert(v)
    next unless v.class == value.class
    return true if value == v
  end
  return false
end
