def % *a, &b
  a.flatten!

  string = case a.last
  when Hash
    expand a.pop
  else
    self
  end

  if a.empty?
    string
  else
    Percent.bind(string).call(a, &b)
  end

end

def expand! vars = {}
  loop do
    changed = false
    vars.each do |var, value|
      var = var.to_s
      var.gsub! /[^a-zA-Z0-9_]/, ''
      [
        /\%\{#{ var }\}/,
      ].each do |pat|
        changed = gsub! pat, "#{ value }"
      end
    end
    break unless changed
  end
  self
end
